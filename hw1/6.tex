\item[\text{в)}] $u_{t}= \Delta u + x^{2}y^{2}\sin t, \: t>0 , \: (x,y,z) \in \mathbb{R}^{3}$ \\
  $u|_{t=0}=xyz\cos x$ \\
  \begin{gather*}
    u = v + w \\
    1)\begin{cases}
      v_{t} = \Delta v + x^{2}y^{2}\sin t \\ v|_{t=0} = 0
    \end{cases} \qquad 2) \begin{cases}
      w_{t} = \Delta w \\ w|_{t=0} = xyz\cos x 
    \end{cases} \\
    1) v = h_{1}x^{2}y^{2} + (x^{2}+y^{2})h_{2}(t) + h_{3}(t) \\
    x^{2}y^{2}h_{1}' + (x^{2}+y^{2})h_{2}'+h_{3}' = 2(x^{2}+y^{2})h_{1}+4h_{2}+x^{2}y^{2}\sin t \\
    \begin{cases}
      h_{1}' = \sin t \\ h_{2}' = 2h_{1} \\ h_{3}' = 4h_{2} 
    \end{cases} \\
    h_{1} = -\cos t + C \quad h(0) = 0 \Rightarrow C = 1 \\
    h_{1} = 1 - \cos t \\
    h_{2} = 2t - 2\sin t + C \quad h_{2}(0) = 0 \Rightarrow C = 0 \\
    h_{3} = 4t^{2} +8\cos t + C \quad h_{3}(0) = 0 \Rightarrow C = -8 \\
  \end{gather*} \newpage
  \begin{gather*}
    v = x^{2}y^{2}(1-\cos t) + (x^{2}+y^{2})2(t-\sin t) +4t^{2}+8\cos t - 8 \\
    2)w=p(x,t)q(y,z,t) \\
    \begin{cases}
      p_{t} = \Delta p \\ p|_{t=0} = x\cos x
    \end{cases} \qquad \begin{cases}
      q_{t} = \Delta q \\ q|_{t=0} = yz 
    \end{cases} \\
    q = q|_{t=0}+tq_{t}|_{t=0} = yz \\
    p = h_{1}(t)x\cos x + h_{2}(t)\sin x \\
    h_{1}'(t)x\cos x +h_{2}'\sin x = h_{1}(-x\cos x -2\sin x) - h_{2}\sin x \\
    \begin{cases}
      h_{1}'(t) = -h_{1} \\ h_{2}'(t) = -2h_{1}-h_{2} \\
    \end{cases} \\
    h_{1} = Ce^{-t} \quad h_{1}(0) = 1 \Rightarrow C = 1 \\
    h_{1} = e^{-t} \\
    h_{2}' = -2e^{-t} - h_{2} \: \Rightarrow \: h_{2} = Ae^{-t} - 2te^{-t} \\
    h_{2}(0) = 0 \Rightarrow A = 0 \\
    h_{2} = -2te^{-t} \\
    \boxed{
      \begin{split}
        u &= (-2te^{-t}\sin x + e^{-t}x\cos x)yz +x^{2}y^{2}(1-\cos t) \ + \\
          &+(x^{2}+y^{2})2(t-\sin t) +4t^{2}+8\cos t - 8
    \end{split}}
  \end{gather*}
